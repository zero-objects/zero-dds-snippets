## 8. Exact extraction recipe (per submessage)
  # submessage distribution
  tshark -r <pcap> -Y rtps -T fields -e rtps.sm.id | tr ',\t' '\n\n' | sort | uniq -c
  # representative frame of a submessage type (filter by sm.id + writer entity)
  #   SPDP DATA      : rtps.sm.wrEntityId==0x000100c2 && rtps.sm.id==0x15
  #   SEDP pub DATA  : rtps.sm.wrEntityId==0x000003c2 && rtps.sm.id==0x15
  #   user DATA      : rtps.sm.wrEntityId==0x00000102 && rtps.sm.id==0x15
  #   HEARTBEAT/ACKNACK/GAP/INFO_TS/INFO_DST/DATA_FRAG: rtps.sm.id==0x07/06/08/09/0e/16
  # NOTE: `tshark -c 1` + display filter returns EMPTY on these captures (known
  #       caveat). Resolve frame.number first, then filter on it:
  fn=$(tshark -r <pcap> -Y '<filter>' -T fields -e frame.number | head -1)
  tshark -r <pcap> -Y "frame.number==$fn" -O rtps -V   # field tree
  tshark -r <pcap> -Y "frame.number==$fn" -x           # raw hex
  # SPDP/SEDP PID order
  tshark -r <pcap> -Y 'rtps.sm.wrEntityId==0x000100c2' -O rtps -V -c 1 | grep -E 'parameterId|PID_'
